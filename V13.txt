        function formatCPF(input) {
            let value = input.value.replace(/\D/g, '');
            value = value.replace(/(\d{3})(\d)/, '$1.$2');
            value = value.replace(/(\d{3})(\d)/, '$1.$2');
            value = value.replace(/(\d{3})(\d{1,2})$/, '$1-$2');
            input.value = value;
        }

        function formatWhatsApp(input) {
            let value = input.value.replace(/\D/g, '');
            if (value.length <= 2) input.value = value;
            else if (value.length <= 7) input.value = `(${value.slice(0, 2)}) ${value.slice(2)}`;
            else input.value = `(${value.slice(0, 2)}) ${value.slice(2, 7)}-${value.slice(7, 11)}`;
        }

        function formatCurrency(input) {
            let value = input.value.replace(/\D/g, '');
            value = (parseInt(value || 0) / 100).toFixed(2);
            input.value = value.replace('.', ',').replace(/(\d)(?=(\d{3})+(?!\d))/g, '$1.');
        }

        function parseCurrency(value) {
            return parseFloat(value.replace(/\./g, '').replace(',', '.')) || 0;
        }

        function formatMoney(value) {
            return 'R$ ' + value.toFixed(2).replace('.', ',').replace(/(\d)(?=(\d{3})+(?!\d))/g, '$1.');
        }

        function formatDate(dateString) {
            const date = new Date(dateString + 'T00:00:00');
            return date.toLocaleDateString('pt-BR');
        }

        function calculateBalanceUpToMonth(year, month, accountId = null) {
            let totalIncome = 0;
            let totalExpense = 0;
            
            transactions.forEach(t => {
                if (accountId && t.account !== accountId) return;
                
                const tYear = t.year;
                const tMonth = t.month;
                
                if (tYear < year || (tYear === year && tMonth <= month)) {
                    if (t.type === 'income') {
                        totalIncome += t.amount;
                    } else {
                        totalExpense += t.amount;
                    }
                }
            });
            
            return totalIncome - totalExpense;
        }

        function calculateGoalsBalance() {
            return goals.reduce((sum, g) => sum + (g.current || 0), 0);
        }

        function calculateAccountTotals(accountId, year, month) {
            let income = 0;
            let expense = 0;
            
            transactions.forEach(t => {
                if (t.account === accountId && t.year === year && t.month === month) {
                    if (t.type === 'income') {
                        income += t.amount;
                    } else {
                        expense += t.amount;
                    }
                }
            });
            
            return { income, expense, balance: income - expense };
        }

        // LOAD DATA - COM ALERTA DE DESPESAS SEM SALDO
        // ============================================
        function loadData() {
            const currentMonth = currentDate.getMonth();
            const currentYear = currentDate.getFullYear();
            
            // Filtrar transações do mês atual (com filtro de conta se ativo)
            let monthTransactions = transactions.filter(t => {
                const matchesMonth = t.month === currentMonth && t.year === currentYear;
                const matchesAccount = currentAccountFilter ? t.account === currentAccountFilter : true;
                return matchesMonth && matchesAccount;
            });
            
            // Calcular saldo acumulado até o mês anterior (com filtro de conta)
            const prevMonth = new Date(currentDate);
            prevMonth.setMonth(prevMonth.getMonth() - 1);
            const prevMonthBalance = calculateBalanceUpToMonth(prevMonth.getFullYear(), prevMonth.getMonth(), currentAccountFilter);
            
            // Mostrar/esconder saldo do mês anterior
            if (prevMonthBalance !== 0) {
                document.getElementById('previousBalance').style.display = 'flex';
                document.getElementById('previousBalanceValue').textContent = formatMoney(prevMonthBalance);
            } else {
                document.getElementById('previousBalance').style.display = 'none';
            }
            
            // Calcular totais
            const income = monthTransactions.filter(t => t.type === 'income').reduce((sum, t) => sum + t.amount, 0);
            const expense = monthTransactions.filter(t => t.type === 'expense').reduce((sum, t) => sum + t.amount, 0);
            const goalsTotal = calculateGoalsBalance();
            const accumulatedBalance = calculateBalanceUpToMonth(currentYear, currentMonth, currentAccountFilter);
            
            // Atualizar UI
            document.getElementById('totalIncome').textContent = formatMoney(income);
            document.getElementById('totalExpense').textContent = formatMoney(expense);
            document.getElementById('totalBalance').textContent = formatMoney(accumulatedBalance);
            document.getElementById('goalsSummaryValue').textContent = formatMoney(goalsTotal) + ' guardados';
            
            updateSmartAlert(income, expense, goalsTotal, accumulatedBalance);
            renderAccountsList();
            updateAccountsDisplay();
            
            // Mostrar info de filtro de conta
            if (currentAccountFilter) {
                const acc = accounts[currentAccountFilter];
                document.getElementById('accountFilterInfo').style.display = 'flex';
                document.getElementById('accountFilterText').textContent = 'Filtrando por: ' + (acc ? acc.name : 'Conta');
            } else {
                document.getElementById('accountFilterInfo').style.display = 'none';
            }
            
            // Renderizar transações e gráfico com filtros aplicados
            renderTransactions(monthTransactions, accumulatedBalance);
            updateChart(monthTransactions, currentChartFilter);
        }

        function updateSmartAlert(income, expense, goalsTotal, balance) {
            const alertBox = document.getElementById('smartAlert');
            const alertTitle = document.getElementById('alertTitle');
            const alertText = document.getElementById('alertText');
            
            if (income === 0 && expense === 0) {
                alertBox.className = 'smart-alert warning';
                alertTitle.textContent = '👋 Bem-vindo!';
                alertText.textContent = 'Comece adicionando suas receitas e despesas';
                return;
            }
            
            if (income > 0) {
                const percent = (expense / income) * 100;
                
                if (percent < 70) {
                    alertBox.className = 'smart-alert success';
                    alertTitle.textContent = '🎉 Ótimo progresso!';
                    alertText.textContent = `Você está gastando apenas ${percent.toFixed(0)}% da sua renda. Continue assim!`;
                } else if (percent >= 70 && percent < 80) {
                    alertBox.className = 'smart-alert warning';
                    alertTitle.textContent = '⚠️ Atenção!';
                    alertText.textContent = `Você já usou ${percent.toFixed(0)}% da renda. Fique de olho nos gastos!`;
                } else if (percent >= 80 && percent < 100) {
                    alertBox.className = 'smart-alert orange';
                    alertTitle.textContent = '🚨 Cuidado!';
                    alertText.textContent = `Gastos excessivos! Você já usou ${percent.toFixed(0)}% da renda.`;
                } else {
                    alertBox.className = 'smart-alert danger';
                    alertTitle.textContent = '❌ Precisamos melhorar!';
                    alertText.textContent = 'Sua gestão financeira precisa de atenção. Gastos acima da renda!';
                }
            } else if (balance < 0) {
                alertBox.className = 'smart-alert danger';
                alertTitle.textContent = '💸 Saldo Negativo';
                alertText.textContent = 'Você está no vermelho. É hora de reorganizar suas finanças!';
            }
        }

        function calculateAccountBalance(accountId) {
            const acc = accounts[accountId];
            if (!acc) return 0;
            
            let balance = acc.balance || 0;
            
            transactions.forEach(t => {
                if (t.account === accountId) {
                    if (t.type === 'income') {
                        balance += t.amount;
                    } else {
                        balance -= t.amount;
                    }
                }
            });
            
            return balance;
        }

        // ============================================
        // TRANSAÇÕES - COM TRIÂNGULOS AMARELOS E ORDENAÇÃO
        // ============================================
        function renderTransactions(monthTransactions, accumulatedBalance) {
            const container = document.getElementById('recentTransactions');
            
            // Aplicar filtro de tipo (tudo/entradas/saídas)
            let filteredTransactions = monthTransactions;
            if (currentTransactionFilter !== 'all') {
                filteredTransactions = monthTransactions.filter(t => t.type === currentTransactionFilter);
            }
            
            // Verificar despesas sem saldo (para alerta visual)
            const currentBalance = accumulatedBalance;
            
            // Marcar despesas que excedem o saldo disponível
            filteredTransactions.forEach(t => {
                t.isPending = false; // Resetar
                if (t.type === 'expense') {
                    // Se a despesa for maior que o saldo acumulado, marcar como pendente/alerta
                    if (t.amount > currentBalance && currentBalance > 0) {
                        t.isPending = true;
                    } else if (currentBalance <= 0) {
                        t.isPending = true;
                    }
                }
            });
            
            // ORDENAÇÃO: Despesas pendentes primeiro, depois por tipo (receitas antes), depois por data
            filteredTransactions.sort((a, b) => {
                // Primeiro: despesas pendentes (sem saldo) vêm primeiro
                if (a.isPending && !b.isPending) return -1;
                if (!a.isPending && b.isPending) return 1;
                
                // Segundo: ordenar por tipo (income vem antes de expense)
                if (a.type !== b.type) {
                    return a.type === 'income' ? -1 : 1;
                }
                
                // Terceiro: ordenar por data (mais recente primeiro)
                const dateA = new Date(a.date + 'T00:00:00');
                const dateB = new Date(b.date + 'T00:00:00');
                return dateB - dateA;
            });
            
            if (filteredTransactions.length === 0) {
                container.innerHTML = `
                    <div class="empty-state">
                        <i class="fas fa-receipt"></i>
                        <p>Nenhum lançamento</p>
                    </div>
                `;
                return;
            }
            
            let html = '';
            filteredTransactions.forEach(t => {
                const category = getCategory(t.type, t.category);
                const icon = category ? category.icon : (t.type === 'income' ? 'fa-arrow-down' : 'fa-arrow-up');
                const color = category ? category.color : (t.type === 'income' ? '#10b981' : '#ef4444');
                const accountName = t.account && accounts[t.account] ? accounts[t.account].name : '';
                
                // Adicionar classe pending se for despesa sem saldo
                const pendingClass = t.isPending ? 'pending' : '';
                const pendingBadge = t.isPending ? '<div class="pending-badge"><i class="fas fa-exclamation"></i></div>' : '';
                
                html += `
                    <div class="transaction-item ${pendingClass}" data-id="${t.id}">
                        ${pendingBadge}
                        <div class="transaction-icon ${t.type}" style="background: ${color}20; color: ${color};">
                            <i class="fas ${icon}"></i>
                        </div>
                        <div class="transaction-info">
                            <div class="transaction-title">${t.description || category?.name || 'Sem categoria'}</div>
                            <div class="transaction-meta">
                                ${formatDate(t.date)} ${accountName ? '• ' + accountName : ''}
                            </div>
                        </div>
                        <div class="transaction-amount ${t.type}">
                            ${t.type === 'income' ? '+' : '-'} ${formatMoney(t.amount)}
                        </div>
                        <div class="transaction-actions-row">
                            <button class="tx-action-btn edit" onclick="editTransaction('${t.id}')">
                                <i class="fas fa-edit"></i>
                            </button>
                            <button class="tx-action-btn delete" onclick="deleteTransaction('${t.id}')">
                                <i class="fas fa-trash"></i>
                            </button>
                        </div>
                    </div>
                `;
            });
            
            container.innerHTML = html;
            
            // Atualizar cor da caixa de saídas se houver despesas pendentes
            const hasPendingExpenses = filteredTransactions.some(t => t.isPending);
            const expenseStat = document.querySelector('.balance-stat:last-child .stat-icon');
            if (hasPendingExpenses && expenseStat) {
                expenseStat.style.background = 'rgba(249, 115, 22, 0.3)';
                expenseStat.style.color = '#c2410c';
            } else if (expenseStat) {
                expenseStat.style.background = 'rgba(15,23,42,0.1)';
                expenseStat.style.color = 'var(--navy)';
            }
        }

        function updateChart(monthTransactions, filter = 'all') {
            currentChartFilter = filter;
            
            // Atualizar botões de filtro do gráfico
            document.querySelectorAll('.chart-filter').forEach((btn, index) => {
                btn.classList.remove('active');
                if ((filter === 'all' && index === 0) || 
                    (filter === 'income' && index === 1) || 
                    (filter === 'expense' && index === 2)) {
                    btn.classList.add('active');
                }
            });
            
            // Filtrar transações para o gráfico
            let chartTransactions = monthTransactions;
            if (filter !== 'all') {
                chartTransactions = monthTransactions.filter(t => t.type === filter);
            }
            
            const ctx = document.getElementById('expenseChart').getContext('2d');
            
            // Agrupar por categoria
            const categories = {};
            chartTransactions.forEach(t => {
                const catName = t.category || 'Outros';
                if (!categories[catName]) {
                    categories[catName] = { amount: 0, type: t.type };
                }
                categories[catName].amount += t.amount;
            });
            
            const labels = Object.keys(categories);
            const data = labels.map(l => categories[l].amount);
            const colors = labels.map(l => {
                const cat = getCategory(categories[l].type, l);
                return cat ? cat.color : (categories[l].type === 'income' ? '#10b981' : '#ef4444');
            });
            
            if (expenseChart) {
                expenseChart.destroy();
            }
            
            if (labels.length === 0) {
                // Gráfico vazio
                expenseChart = new Chart(ctx, {
                    type: 'doughnut',
                    data: {
                        labels: ['Sem dados'],
                        datasets: [{
                            data: [1],
                            backgroundColor: ['#e2e8f0'],
                            borderWidth: 0
                        }]
                    },
                    options: {
                        responsive: true,
                        maintainAspectRatio: false,
                        plugins: {
                            legend: { display: false }
                        }
                    }
                });
                return;
            }
            
            expenseChart = new Chart(ctx, {
                type: 'doughnut',
                data: {
                    labels,
                    datasets: [{
                        data,
                        backgroundColor: colors,
                        borderWidth: 2,
                        borderColor: '#fff'
                    }]
                },
                options: {
                    responsive: true,
                    maintainAspectRatio: false,
                    plugins: {
                        legend: {
                            position: 'bottom',
                            labels: {
                                padding: 15,
                                usePointStyle: true,
                                font: { size: 11 }
                            }
                        }
                    }
                }
            });
        }

        // CORREÇÃO: Handler separado para o formulário de transação
        async function handleTransactionSubmit(event) {
            event.preventDefault();
            
            const amount = parseCurrency(document.getElementById('amount').value);
            const date = document.getElementById('date').value;
            const account = document.getElementById('account').value;
            const category = document.getElementById('selectedCategory').value;
            const description = document.getElementById('description').value.trim();
            
            if (!amount || amount <= 0) {
                showToast('Digite um valor válido', 'error');
                return;
            }
            
            if (!category) {
                showToast('Selecione uma categoria', 'error');
                return;
            }
            
            const dateObj = new Date(date + 'T00:00:00');
            const transaction = {
                id: editingTransaction ? editingTransaction.id : 'tx_' + Date.now(),
                type: currentType,
                amount,
                date,
                month: dateObj.getMonth(),
                year: dateObj.getFullYear(),
                account,
                category,
                description,
                createdAt: editingTransaction ? editingTransaction.createdAt : Date.now()
            };
            
            if (editingTransaction) {
                const index = transactions.findIndex(t => t.id === editingTransaction.id);
                if (index !== -1) transactions[index] = transaction;
                showToast('Transação atualizada!');
            } else {
                transactions.push(transaction);
                showToast('Transação salva!');
            }
            
            // Fechar modal e limpar formulário
            closeModal('transactionModal');
            
            // Salvar no Firebase e recarregar dados
            await saveToFirebase();
            loadData();
        }

        function editTransaction(id) {
            const t = transactions.find(tx => tx.id === id);
            if (!t) return;
            
            editingTransaction = t;
            currentType = t.type;

        async function handleGoalSubmit(event) {
            event.preventDefault();
            
            const name = document.getElementById('goalName').value.trim();
            const target = parseCurrency(document.getElementById('goalTarget').value);
            const current = parseCurrency(document.getElementById('goalCurrent').value) || 0;
            const date = document.getElementById('goalDate').value;
            
            if (!name || !target) {
                showToast('Preencha nome e valor objetivo', 'error');
                return;
            }
            
            const goal = {
                id: editingGoal ? editingGoal.id : 'goal_' + Date.now(),
                name,
                target,
                current,
                date,
                createdAt: editingGoal ? editingGoal.createdAt : Date.now()
            };
            
            if (editingGoal) {
                const index = goals.findIndex(g => g.id === editingGoal.id);
                if (index !== -1) goals[index] = goal;
                showToast('Meta atualizada!');
            } else {
                goals.push(goal);
                showToast('Meta criada!');
            }
            
            closeModal('goalModal');
            await saveToFirebase();
            renderGoals();
            loadData();
        }

        let reportChart = null;

        function updateReportChart() {
            const currentMonth = currentDate.getMonth();
            const currentYear = currentDate.getFullYear();
            
            const monthTransactions = transactions.filter(t => 
                t.month === currentMonth && t.year === currentYear
            );
            
            const income = monthTransactions.filter(t => t.type === 'income').reduce((sum, t) => sum + t.amount, 0);
            const expense = monthTransactions.filter(t => t.type === 'expense').reduce((sum, t) => sum + t.amount, 0);
            const savings = income > 0 ? ((income - expense) / income * 100) : 0;
            
            document.getElementById('reportIncome').textContent = formatMoney(income);
            document.getElementById('reportExpense').textContent = formatMoney(expense);
            document.getElementById('reportSavings').textContent = savings.toFixed(0) + '%';
            
            // Listas de entradas e saídas
            const incomeList = document.getElementById('incomeList');
            const expenseList = document.getElementById('expenseList');
            
            const incomeTx = monthTransactions.filter(t => t.type === 'income');
            const expenseTx = monthTransactions.filter(t => t.type === 'expense');
            
            incomeList.innerHTML = incomeTx.length ? incomeTx.map(t => `
                <div class="report-item">
                    <span>${t.description || t.category}</span>
                    <span style="color: var(--success); font-weight: 600;">+ ${formatMoney(t.amount)}</span>
                </div>
            `).join('') : '<p style="color: var(--gray-400); text-align: center;">Nenhuma entrada</p>';
            
            expenseList.innerHTML = expenseTx.length ? expenseTx.map(t => `
                <div class="report-item">
                    <span>${t.description || t.category}</span>
                    <span style="color: var(--danger); font-weight: 600;">- ${formatMoney(t.amount)}</span>
                </div>
            `).join('') : '<p style="color: var(--gray-400); text-align: center;">Nenhuma saída</p>';
            
            // Gráfico de barras
            const ctx = document.getElementById('reportChart').getContext('2d');
            
            if (reportChart) {
                reportChart.destroy();
            }
            
            reportChart = new Chart(ctx, {
                type: 'bar',
                data: {
                    labels: ['Entradas', 'Saídas', 'Saldo'],
                    datasets: [{
                        data: [income, expense, income - expense],
                        backgroundColor: ['#10b981', '#ef4444', income - expense >= 0 ? '#3b82f6' : '#f59e0b'],
                        borderRadius: 8
                    }]
                },
                options: {
                    responsive: true,
                    maintainAspectRatio: false,
                    plugins: {
                        legend: { display: false }
                    },
                    scales: {
                        y: {
                            beginAtZero: true,
                            ticks: {
                                callback: function(value) {
                                    return 'R$ ' + value.toFixed(0);
                                }
                            }
                        }
                    }
                }
            });
        }

        // PDF - ORDENADO POR DATA CRESCENTE COM TOTAL DESTACADO
        // ============================================
        function generatePDF() {
            const { jsPDF } = window.jspdf;
            const doc = new jsPDF();
            
            const currentMonth = currentDate.getMonth();
            const currentYear = currentDate.getFullYear();
            const months = ['Janeiro', 'Fevereiro', 'Março', 'Abril', 'Maio', 'Junho', 
                          'Julho', 'Agosto', 'Setembro', 'Outubro', 'Novembro', 'Dezembro'];
            
            // Filtrar transações do mês (com filtro de conta se ativo)
            let monthTransactions = transactions.filter(t => {
                const matchesMonth = t.month === currentMonth && t.year === currentYear;
                const matchesAccount = currentAccountFilter ? t.account === currentAccountFilter : true;
                return matchesMonth && matchesAccount;
            });
            
            // ORDENAÇÃO CRESCENTE POR DATA (mais antiga primeiro)
            monthTransactions.sort((a, b) => {
                const dateA = new Date(a.date + 'T00:00:00');
                const dateB = new Date(b.date + 'T00:00:00');
                return dateA - dateB; // Crescente
            });
            
            const income = monthTransactions.filter(t => t.type === 'income').reduce((sum, t) => sum + t.amount, 0);
            const expense = monthTransactions.filter(t => t.type === 'expense').reduce((sum, t) => sum + t.amount, 0);
            const balance = income - expense;
            
            // Cabeçalho
            doc.setFillColor(15, 23, 42);
            doc.rect(0, 0, 210, 40, 'F');
            
            doc.setTextColor(212, 175, 55);
            doc.setFontSize(24);
            doc.text('Minhas Finanças PRO', 105, 20, { align: 'center' });
            
            doc.setTextColor(255, 255, 255);
            doc.setFontSize(12);
            doc.text(`Relatório de ${months[currentMonth]} de ${currentYear}`, 105, 32, { align: 'center' });
            
            if (currentAccountFilter && accounts[currentAccountFilter]) {
                doc.text(`Conta: ${accounts[currentAccountFilter].name}`, 105, 38, { align: 'center' });
            }
            
            // Resumo
            doc.setTextColor(15, 23, 42);
            doc.setFontSize(16);
            doc.text('Resumo do Mês', 20, 55);
            
            doc.setFontSize(12);
            doc.text(`Total de Entradas: ${formatMoney(income)}`, 20, 65);
            doc.text(`Total de Saídas: ${formatMoney(expense)}`, 20, 72);
            doc.text(`Saldo: ${formatMoney(balance)}`, 20, 79);
            
            // Transações
            doc.setFontSize(16);
            doc.text('Lançamentos', 20, 95);
            
            let y = 105;
            doc.setFontSize(10);
            
            // Separar por tipo e ordenar
            const incomeItems = monthTransactions.filter(t => t.type === 'income');
            const expenseItems = monthTransactions.filter(t => t.type === 'expense');
            
            // Entradas
            if (incomeItems.length > 0) {
                doc.setTextColor(16, 185, 129);
                doc.setFontSize(12);
                doc.text('ENTRADAS:', 20, y);
                y += 8;
                doc.setFontSize(10);
                
                incomeItems.forEach(t => {
                    if (y > 270) {
                        doc.addPage();
                        y = 20;
                    }
                    
                    const category = getCategory(t.type, t.category);
                    
                    doc.setTextColor(16, 185, 129);
                    doc.text(`+ ${formatMoney(t.amount)}`, 190, y, { align: 'right' });
                    
                    doc.setTextColor(15, 23, 42);
                    doc.text(`${formatDate(t.date)} - ${t.description || category?.name || 'Sem categoria'}`, 20, y);
                    
                    y += 7;
                });
                y += 5;
            }
            
            // Saídas
            if (expenseItems.length > 0) {
                if (y > 250) {
                    doc.addPage();
                    y = 20;
                }
                
                doc.setTextColor(239, 68, 68);
                doc.setFontSize(12);
                doc.text('SAÍDAS:', 20, y);
                y += 8;
                doc.setFontSize(10);
                
                expenseItems.forEach(t => {
                    if (y > 270) {
                        doc.addPage();
                        y = 20;
                    }
                    
                    const category = getCategory(t.type, t.category);
                    
                    doc.setTextColor(239, 68, 68);
                    doc.text(`- ${formatMoney(t.amount)}`, 190, y, { align: 'right' });
                    
                    doc.setTextColor(15, 23, 42);
                    doc.text(`${formatDate(t.date)} - ${t.description || category?.name || 'Sem categoria'}`, 20, y);
                    
                    y += 7;
                });
                y += 10;
            }
            
            // TOTAL DESTACADO NO FINAL
            if (y > 250) {
                doc.addPage();
                y = 30;
            }
            
            // Linha separadora dourada
            doc.setDrawColor(212, 175, 55);
            doc.setLineWidth(0.5);
            doc.line(20, y, 190, y);
            y += 10;
            
            // Box azul marinho com total
            doc.setFillColor(15, 23, 42);
            doc.roundedRect(20, y, 170, 25, 3, 3, 'F');
            
            doc.setTextColor(212, 175, 55);
            doc.setFontSize(11);
            doc.text('TOTAL DE SAÍDAS DO MÊS:', 105, y + 10, { align: 'center' });
            
            doc.setTextColor(255, 255, 255);
            doc.setFontSize(14);
            doc.setFont(undefined, 'bold');
            doc.text(formatMoney(expense), 105, y + 20, { align: 'center' });
            
            // Categoria mais gastadora
            if (expenseItems.length > 0) {
                y += 35;
                const categories = {};
                expenseItems.forEach(t => {
                    const cat = t.category || 'Outros';
                    categories[cat] = (categories[cat] || 0) + t.amount;
                });
                
                const topCategory = Object.keys(categories).reduce((a, b) => categories[a] > categories[b] ? a : b);
                
                doc.setTextColor(15, 23, 42);
                doc.setFontSize(10);
                doc.setFont(undefined, 'normal');
                doc.text(`Maior gasto: ${topCategory} (${formatMoney(categories[topCategory])})`, 105, y, { align: 'center' });
            }
            
            // Rodapé
            const pageCount = doc.internal.getNumberOfPages();
            for (let i = 1; i <= pageCount; i++) {
                doc.setPage(i);
                doc.setFontSize(8);
                doc.setTextColor(128, 128, 128);
                doc.text('Minhas Finanças PRO - ASM Gestão Financeira', 105, 290, { align: 'center' });
            }
            
            doc.save(`financas_${months[currentMonth]}_${currentYear}.pdf`);
            showToast('PDF gerado com sucesso!');
        }
